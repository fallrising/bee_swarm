#### check_environment.py

"""
Áí∞Â¢ÉÈ©óË≠âËÖ≥Êú¨
"""
import sys
import subprocess
import importlib
from pathlib import Path
from typing import List, Dict, Any

class EnvironmentChecker:
    def __init__(self):
        self.results: List[Dict[str, Any]] = []
    
    def check_python_version(self):
        """Ê™¢Êü• Python ÁâàÊú¨"""
        version = sys.version_info
        required = (3, 8)
        
        if version >= required:
            self.results.append({
                "component": "Python",
                "status": "‚úÖ PASS",
                "message": f"Version {version.major}.{version.minor}.{version.micro}"
            })
        else:
            self.results.append({
                "component": "Python",
                "status": "‚ùå FAIL",
                "message": f"Version {version.major}.{version.minor} < {required[0]}.{required[1]}"
            })
    
    def check_required_packages(self):
        """Ê™¢Êü•ÂøÖË¶ÅÁöÑ Python Â•ó‰ª∂"""
        required_packages = [
            "fastapi", "uvicorn", "simpy", "pandas", 
            "numpy", "matplotlib", "sqlalchemy", "redis"
        ]
        
        for package in required_packages:
            try:
                importlib.import_module(package)
                self.results.append({
                    "component": f"Package: {package}",
                    "status": "‚úÖ PASS",
                    "message": "Installed"
                })
            except ImportError:
                self.results.append({
                    "component": f"Package: {package}",
                    "status": "‚ùå FAIL",
                    "message": "Not installed"
                })
    
    def check_external_commands(self):
        """Ê™¢Êü•Â§ñÈÉ®ÂëΩ‰ª§"""
        commands = ["git", "docker", "docker-compose"]
        
        for cmd in commands:
            try:
                result = subprocess.run(
                    [cmd, "--version"], 
                    capture_output=True, 
                    text=True, 
                    timeout=10
                )
                if result.returncode == 0:
                    version = result.stdout.strip().split('\n')[0]
                    self.results.append({
                        "component": cmd.capitalize(),
                        "status": "‚úÖ PASS",
                        "message": version
                    })
                else:
                    self.results.append({
                        "component": cmd.capitalize(),
                        "status": "‚ùå FAIL",
                        "message": "Command failed"
                    })
            except (subprocess.TimeoutExpired, FileNotFoundError):
                self.results.append({
                    "component": cmd.capitalize(),
                    "status": "‚ùå FAIL",
                    "message": "Not found"
                })
    
    def check_environment_variables(self):
        """Ê™¢Êü•Áí∞Â¢ÉËÆäÊï∏"""
        import os
        required_vars = ["ENVIRONMENT"]
        optional_vars = ["DATABASE_URL", "REDIS_URL", "SECRET_KEY"]
        
        for var in required_vars:
            if os.getenv(var):
                self.results.append({
                    "component": f"ENV: {var}",
                    "status": "‚úÖ PASS",
                    "message": "Set"
                })
            else:
                self.results.append({
                    "component": f"ENV: {var}",
                    "status": "‚ùå FAIL",
                    "message": "Not set"
